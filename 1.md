

## 使用Google Antigravity

**本作業的程式碼與結構是透過 Google Antigravity生成的。**



## 功能重點整理
- **基礎與進階要求**
  - **單一輸入區**：將兩個輸入框改為仿照手機/電腦小算盤的單一 TextBox (本專案使用 TextBlock 達更佳排版效果)。
  - **直覺介面**：提供完整的 0-9 數字鍵與四則運算符號。
  - **回刪功能 (Backspace)**：允許刪除最後輸入的單一字元，避免填錯需要全部重來(`⌫` 鍵)。
  - **清除功能 (C/CE)**：提供 C (清除所有狀態) 與 CE (僅清除當前輸入值)。
- **加分項目 (Bonus)**
  - **介面美化**：整體採用 Modern UI 深色主題，按鈕具備隨滑鼠互動的變色動畫 (Hover & Pressed 觸發器)。
  - **鍵盤支援**：除了滑鼠點擊外，更支援直接使用實體鍵盤的數字區 (NumPad)、四則運算符號、Enter (等號)、Backspace、Escape (清除) 進行純鍵盤操作。
  - **進階運算**：結合常用的計算機按鈕，支援了倒數 (1/x)、平方 (x²)、開根號 (√x) 與百分比 (%) 的快捷轉換。
- **例外與防呆處理 (防 Bug)**
  - 當使用者觸發「除以零」或「對負數開根號」時，系統不會崩潰跳錯，而是攔截 `DivideByZeroException` 等錯誤，在畫面上友善提示「無法除以零」。
  - 防呆機制包含：避免輸入超過兩個小數點、限制輸入長度。

 ```xaml.cs            
// previousValue: 用來儲存運算符號左邊的數值 (例如 A + B 中的 A)
private double previousValue = 0;

// currentOperator: 用來儲存當前點選的運算符號 (例如 +, -, *, /)
private string currentOperator = "";

// isNewInput: 狀態旗標。判斷下次點擊數字時，是否應該重新開始輸入新數字 (取代目前畫面的數字)
private bool isNewInput = true;

// isErrorState: 錯誤旗標。用來處理像是「除以零」的例外狀況，避免程式崩潰
private bool isErrorState = false;
```

**核心按鈕事件處理**

 ```xaml.cs            
/// <summary>
/// 處理所有數字按鍵 (0~9) 的點擊事件
/// </summary>
private void Number_Click(object sender, RoutedEventArgs e)
{
    // 如果當前是錯誤狀態 (例如剛發生除以零)，則必須先清除全部才能繼續
    if (isErrorState) ClearAllState();

    // 取得觸發事件的按鈕文字 (例如 "1"、"2")
    Button btn = sender as Button;
    string number = btn.Content.ToString();

    // 如果是新輸入階段，或目前畫面只有 "0"，則直接用新數字取代畫面
    if (isNewInput || DisplayText.Text == "0")
    {
        DisplayText.Text = number;
        isNewInput = false; // 取代後，接下來的數字都要用「附加」的方式
    }
    else
    {
        // 如果目前數字太長則防止無意義的繼續輸入 (簡易長度防護)
        if (DisplayText.Text.Length < 16)
        {
            DisplayText.Text += number; // 附加數字到後方
        }
    }
}

/// <summary>
/// 處理小數點按鈕 (".") 的點擊事件
/// </summary>
private void Dot_Click(object sender, RoutedEventArgs e)
{
    if (isErrorState) ClearAllState();

    // 如果是剛輸入完運算符號而要開始新數字，按下小數點應該自動補零變成 "0."
    if (isNewInput)
    {
        DisplayText.Text = "0.";
        isNewInput = false;
    }
    // 防呆機制：確保同一個數字中不會出現兩個小數點
    else if (!DisplayText.Text.Contains("."))
    {
        DisplayText.Text += ".";
    }
}

/// <summary>
/// 處理基礎四則運算符號 (+, -, *, /, %) 的點擊事件
/// </summary>
private void Operator_Click(object sender, RoutedEventArgs e)
{
    if (isErrorState) return; // 錯誤狀態下禁止運算

    Button btn = sender as Button;
    string newOperator = btn.Content.ToString();

    // 處理百分比功能 (%)：將目前輸入的值除以 100
    if (newOperator == "%")
    {
        if (double.TryParse(DisplayText.Text, out double currentValue))
        {
            DisplayText.Text = (currentValue / 100).ToString();
        }
        return;
    }

    // 如果之前已經有輸入過運算符號且還沒按等號，先計算前面的暫存結果 (連加/連算邏輯)
    if (!isNewInput && !string.IsNullOrEmpty(currentOperator))
    {
        CalculateResult();
    }

    // 儲存目前顯示的數字為 previousValue (運算元 A)
    if (double.TryParse(DisplayText.Text, out double parsedValue))
    {
        previousValue = parsedValue;
    }

    // 更新當前使用的運算符號
    currentOperator = newOperator;
    isNewInput = true; // 標記為新輸入階段，下一筆數字將會清空畫面輸入

    // 更新歷史輸入顯示，讓使用者知道目前在算什麼 (例如 "123 +")
    HistoryText.Text = $"{previousValue} {currentOperator}";
}

/// <summary>
/// 處理等號 ("=") 的點擊事件，執行最終計算
/// </summary>
private void Equal_Click(object sender, RoutedEventArgs e)
{
    if (isErrorState || string.IsNullOrEmpty(currentOperator)) return;

    // 如果在等號前輸入的是有效數字，則執行計算
    if (!isNewInput)
    {
        CalculateResult();
        HistoryText.Text = "";    // 計算完成後清空歷史紀錄
        currentOperator = "";     // 清空當前運算符號
        isNewInput = true;        // 下次輸入視為全新開始
    }
}
```

**進階加分功能**

 ```xaml.cs            
/// <summary>
/// 處理進階運算符號 (倒數 1/x, 平方 x², 開根號 √x) 的點擊事件
/// </summary>
private void AdvancedOperator_Click(object sender, RoutedEventArgs e)
{
    if (isErrorState) return;

    Button btn = sender as Button;
    string operation = btn.Content.ToString();

    if (double.TryParse(DisplayText.Text, out double currentValue))
    {
        try
        {
            double result = 0;
            if (operation == "1/x")
            {
                // 例外防護：攔截除以零
                if (currentValue == 0) throw new DivideByZeroException();
                result = 1 / currentValue;
                HistoryText.Text = $"1/({currentValue})";
            }
            else if (operation == "x²")
            {
                result = Math.Pow(currentValue, 2);
                HistoryText.Text = $"sqr({currentValue})";
            }
            else if (operation == "√x")
            {
                // 例外防護：開根號不支援負數
                if (currentValue < 0) throw new InvalidOperationException("無效輸入");
                result = Math.Sqrt(currentValue);
                HistoryText.Text = $"√({currentValue})";
            }

            DisplayText.Text = result.ToString();
            isNewInput = true; // 運算完畢後等待下一次輸入
        }
        catch (DivideByZeroException)
        {
            ShowError("無法除以零");
        }
        catch (Exception)
        {
            ShowError("無效輸入");
        }
    }
}

/// <summary>
/// 處理正負號切換 (+/-) 
/// </summary>
private void ToggleSign_Click(object sender, RoutedEventArgs e)
{
    if (isErrorState || DisplayText.Text == "0") return;

    if (DisplayText.Text.StartsWith("-"))
        DisplayText.Text = DisplayText.Text.Substring(1); // 移除負號
    else
        DisplayText.Text = "-" + DisplayText.Text;        // 加入負號
}
```

 ```xaml.cs            
 /// <summary>
 /// 處理「回刪 (Backspace)」: 刪除最後一個輸入的數字
 /// </summary>
 private void Backspace_Click(object sender, RoutedEventArgs e)
 {
     if (isErrorState || isNewInput) return; // 如果準備輸入新數字或錯誤狀態，不處理回刪

     // 如果長度大於1，則刪除最後一個字元
     if (DisplayText.Text.Length > 1)
     {
         DisplayText.Text = DisplayText.Text.Substring(0, DisplayText.Text.Length - 1);
     }
     else
     {
         // 如果只剩最後一個數字，則重置為 "0"
         DisplayText.Text = "0";
         isNewInput = true;
     }
 }
```

**核心計算與共用方法**

 ```xaml.cs            
/// <summary>
/// 根據當前的運算符號 (currentOperator) 執行 A (previousValue) 與 B (當前畫面數字) 的計算
/// </summary>
private void CalculateResult()
{
    if (double.TryParse(DisplayText.Text, out double currentValue))
    {
        double result = 0;
        try
        {
            switch (currentOperator)
            {
                case "+":
                    result = previousValue + currentValue;
                    break;
                case "-":
                    result = previousValue - currentValue;
                    break;
                case "×":
                case "*":
                    result = previousValue * currentValue;
                    break;
                case "÷":
                case "/":
                    // 除以零的例外防護處理 (Exception Handling/ 防 Bug)
                    if (currentValue == 0)
                    {
                        throw new DivideByZeroException();
                    }
                    result = previousValue / currentValue;
                    break;
            }
            DisplayText.Text = result.ToString();
            previousValue = result; // 將計算結果作為下一次運算的基底
        }
        catch (DivideByZeroException)
        {
            ShowError("無法除以零");
        }
        catch (Exception)
        {
            ShowError("發生未知的錯誤");
        }
    }
}

/// <summary>
/// 觸發錯誤狀態，顯示錯誤訊息並鎖定其他功能，直到使用者按清除。
/// </summary>
private void ShowError(string message)
{
    DisplayText.Text = message;
    HistoryText.Text = "";
    isErrorState = true;
}

/// <summary>
/// 回復預設初始狀態 (C)
/// </summary>
private void ClearAllState()
{
    previousValue = 0;
    currentOperator = "";
    isNewInput = true;
    isErrorState = false;
    DisplayText.Text = "0";
    HistoryText.Text = "";
}
```

**鍵盤支援 (進階加分項目)**

 ```xaml.cs            
/// <summary>
/// 處理鍵盤按鍵事件 (KeyDown)，讓使用者可以直接透過實體鍵盤的數字區敲打進行計算
/// </summary>
private void Window_KeyDown(object sender, KeyEventArgs e)
{
    // 防護：若是錯誤狀態下，按下 Escape 直接清除
    if (isErrorState && e.Key != Key.Escape) return;

    // 擷取數字鍵 (主鍵盤 0-9 & 數字鍵盤 NumPad 0-9)
    if (e.Key >= Key.D0 && e.Key <= Key.D9 && Keyboard.Modifiers != ModifierKeys.Shift)
    {
        SimulateButtonClick(((int)e.Key - (int)Key.D0).ToString());
        e.Handled = true;
    }
    else if (e.Key >= Key.NumPad0 && e.Key <= Key.NumPad9)
    {
        SimulateButtonClick(((int)e.Key - (int)Key.NumPad0).ToString());
        e.Handled = true;
    }
    // 處理小數點
    else if (e.Key == Key.Decimal || e.Key == Key.OemPeriod)
    {
        Dot_Click(this, null);
        e.Handled = true;
    }
    // 處理運算符號 (加減乘除)
    else if (e.Key == Key.Add || (e.Key == Key.OemPlus && Keyboard.Modifiers != ModifierKeys.Shift))
    {
        SimulateOperatorClick("+");
        e.Handled = true;
    }
    else if (e.Key == Key.Subtract || e.Key == Key.OemMinus)
    {
        SimulateOperatorClick("-");
        e.Handled = true;
    }
    else if (e.Key == Key.Multiply || (e.Key == Key.D8 && Keyboard.Modifiers == ModifierKeys.Shift))
    {
        SimulateOperatorClick("×");
        e.Handled = true;
    }
    else if (e.Key == Key.Divide || e.Key == Key.OemQuestion)
    {
        SimulateOperatorClick("÷");
        e.Handled = true;
    }
    // 處理等號與 Enter 鍵
    else if (e.Key == Key.Enter || e.Key == Key.Return || (e.Key == Key.OemPlus && Keyboard.Modifiers == ModifierKeys.Shift) || e.Key == Key.OemPlus) // OemPlus without shift might be "=" on some layouts
    {
        // To be robust about whether it's '=' or '+' being pressed: On US layout = is OemPlus,  Shift+= is +
        // But Add is purely '+' from numpad. 
        // Handled simple Enter triggering Equal.
        Equal_Click(this, null);
        e.Handled = true;
    }
    // 處理回刪鍵 (Backspace)
    else if (e.Key == Key.Back)
    {
        Backspace_Click(this, null);
        e.Handled = true;
    }
    // 處理清除鍵 (Escape 映射為 Clear All)
    else if (e.Key == Key.Escape)
    {
        ClearAll_Click(this, null);
        e.Handled = true;
    }
}
```
